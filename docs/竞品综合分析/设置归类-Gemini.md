import pandas as pd

# Load and clean the data
df = pd.read_csv('设置 - Sheet1.csv')
df['导航'] = df['导航'].ffill()
df['功能模块'] = df['功能模块'].ffill()

# Fixed 27 categories
categories = [
    "品牌管理", "门店管理", "Dashboard（主页）", "团队管理", "商品中心",
    "订单中心", "支付中心", "外卖/来取", "营销中心", "促销中心",

    "库存管理中心", "硬件管理中心", "素材管理中心", "供应链中心", "平台业务中心",
    "权限管理中心", "系统设置"
]

def categorize_item(nav, module, setting, desc):
    n, m, s, d = str(nav), str(module), str(setting), str(desc)
    text = (n + m + s + d).lower()

    # Rule-based logic based on common POS patterns
    if any(k in text for k in ['品牌', 'brand']): return "品牌管理"
    if any(k in text for k in ['门店信息', '营业时间', '地址', '店铺']): return "门店管理"
    if any(k in text for k in ['看板', '仪表盘', '核心指标', 'dashboard']): return "Dashboard（主页）"
    if any(k in text for k in ['员工', '工时', '工资', '打卡', '考勤', 'staff']): return "团队管理"
    if any(k in text for k in ['权限', '密码', '角色', '登出', '登录码', '授权']): return "权限管理中心"
    if any(k in text for k in ['商品', '菜单', '分类', '售罄', '规格', '加料', '套餐', '品类', '菜单组', 'item']): return "商品中心"
    if any(k in text for k in ['订单', '退款', '单据', '接单', '取餐码', 'order']): return "订单中心"
    if any(k in text for k in ['支付', '税', '小费', '免单', '抹零', '付款', '付款方式', '结算', 'payment', 'tax', 'tip']): return "支付中心"
    if any(k in text for k in ['外卖', '来取', '配送', '自提', 'delivery', 'pickup']): return "外卖/来取"
    if any(k in text for k in ['营销', '活动', '老客']): return "营销中心"
    if any(k in text for k in ['促销', '折扣', '满减', '优惠券', 'discount']): return "促销中心"
    if any(k in text for k in ['会员', '积分', '等级', 'member']): return "会员中心"
    if any(k in text for k in ['礼品卡', 'gift card']): return "礼品卡中心"
    if any(k in text for k in ['评价', '反馈', 'review']): return "评价中心"
    if any(k in text for k in ['前厅', '桌位', '区域', '茶位', '堂吃']): return "前厅管理中心"
    if any(k in text for k in ['后厨', 'kds', '制作', '出餐']): return "后厨管理中心"
    if any(k in text for k in ['预约', '等位', '预订', 'waitlist']): return "预约等位中心"
    if any(k in text for k in ['报表', '统计', '分析', '数据导出', '报告']): return "报表中心"

    if any(k in text for k in ['库存', '消耗', '入库', 'stock']): return "库存管理中心"
    if any(k in text for k in ['硬件', '钱箱', '叫号屏', '客显', 'kiosk', '扫码枪', '副屏', '设备', 'ip地址']): return "硬件管理中心"

    if any(k in text for k in ['供应链', '采购', '供应商']): return "供应链中心"
    if any(k in text for k in ['平台', '接口', '集成', 'api', '第三方']): return "平台业务中心"

    return "系统设置"

results = {cat: {} for cat in categories}

for _, row in df.iterrows():
    setting = row['功能设置']
    if pd.isna(setting): continue

    cat = categorize_item(row['导航'], row['功能模块'], row['功能设置'], row['设置功能说明'])
    sub_cat = row['功能模块'] if not pd.isna(row['功能模块']) else "通用设置"

    if sub_cat not in results[cat]:
        results[cat][sub_cat] = []
    results[cat][sub_cat].append({
        'name': setting,
        'desc': row['设置功能说明'],
        'type': row['值类型'],
        'value': row['值']
    })

# Format into Markdown

for cat in categories:
    md += f"## {cat}\n"
    if not results[cat]:
        md += "*（当前暂无匹配设置项）*\n\n"
        continue

    for sub_cat, items in results[cat].items():
        md += f"### {sub_cat}\n"
        for item in items:
            val_info = f" `[{item['type']}]`" if not pd.isna(item['type']) else ""
            desc_info = f" — *{item['desc']}*" if not pd.isna(item['desc']) else ""
            md += f"*   **{item['name']}**{val_info}{desc_info}\n"
        md += "\n"

print(md[:5000]) # Check content
